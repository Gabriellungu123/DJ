Primero instalamos el entorno virtual en este caso sera virtualenv:
-
python -m venv venv
--
Luego lo activamos 
-
.\venv\Scripts\Activate

----
Instalar Django
-
pip install django
--
Comprobar de que lo tenemos 
-
django-admin --version
--
Crear Proyecto
django-admin startproject nombre del proyecto .
--
Arrancarlo
-
python manage.py runserver
--
Ahora quierro arrancarlo en el puerto 300
-
python manage.py runserver 3000
-- 
Para iniciar una app
-
python manage.py startapp nombre de la app
#Te abre una carpeta nueva con los archivos .py 
---
Actualizar la base de datos 
-
python manage.py migrate
--
Actualizar la base de datos de una app en especifico
-
python manage.py makemigrations nombre de la app
---
